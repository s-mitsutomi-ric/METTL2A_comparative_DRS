rename_col_and_reshape <- function(df) {
  df |> 
    rename(distance_inches = `Distance_(inches)`) |> 
    pivot_longer(
      cols = -c(distance_inches, sample), 
      names_to = 'channel', values_to = 'intensity'
    )
}
